\begin{align*}
	F_X(x) &= 1 - e^{-\frac{1}{2}\alpha^2x^2} \\
	1 - F &= e^{-\frac{1}{2}\alpha^2x^2} \\
	\ln(1 - F) &= -\frac{1}{2}\alpha^2x^2 \\
	x^2 &= -\frac{2 \ln(1 - F)}{\alpha^2} \\
	x(F) &= \sqrt{-\frac{2 \ln(1 - F)}{\alpha^2}} \\
\end{align*}
